# Without component

This is an example without Component

export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "es":
      return "Hola!";
    case "fr":
      return "Bonjour!";
    case "kr":
      return "안녕하세요!";
    case "zh":
      return "你好!";
    default:
      return "Hello!";
  }
};

## No Canvas

<Story name="NoCanvas">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <>{caption}</>;
  }}
</Story>
